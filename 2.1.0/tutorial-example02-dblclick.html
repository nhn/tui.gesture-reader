            var detect = document.getElementById('detect');
            var direction = document.getElementById('direction');
            var type = document.getElementById('type');

            var info = {};

            var gestureReader = new tui.GestureReader({
                type: 'dblclick',
                clickTerm: 600
            });

            detect.onclick = function(e) {
                e = e || window.event;

                info.x = e.pageX;
                info.y = e.pageY;

                if (gestureReader.isDoubleClick(info)) {
                    type.innerHTML = 'Type : <em>Double Click</em>';
                } else {
                    type.innerHTML = 'Type : <em>Click</em>';
                }
            };
